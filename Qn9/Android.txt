package com.example.qn9;

import android.os.Bundle;
import android.os.Handler;
import android.widget.Button;
import android.widget.ProgressBar;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    ProgressBar progressBar;
    Button buttonStart;
    int progressStatus = 0;
    Handler handler = new Handler();

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        progressBar = findViewById(R.id.progressBar);
        buttonStart = findViewById(R.id.buttonStart);

        buttonStart.setOnClickListener(v -> {
            progressBar.setVisibility(ProgressBar.VISIBLE);
            progressStatus = 0;

            // Run progress in a background thread
            new Thread(() -> {
                while (progressStatus < 100) {
                    progressStatus += 1;
                    try {
                        Thread.sleep(50); // simulate work (50ms per step)
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }

                    // Update progress on the UI thread
                    handler.post(() -> progressBar.setProgress(progressStatus));
                }
            }).start();
        });
    }
}

package com.example.qn9;

import android.hardware.camera2.CameraAccessException;
import android.hardware.camera2.CameraManager;
import android.os.Bundle;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    boolean isOn = false;          // keeps track of flashlight state
    CameraManager cameraManager;   // helps control the phoneâ€™s camera system
    String cameraId;               // identifies which camera to use (usually the back one)

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button btn = findViewById(R.id.button);
        cameraManager = (CameraManager) getSystemService(CAMERA_SERVICE);

        btn.setOnClickListener(v -> {
            try {
                if (!isOn) {
                    cameraId = cameraManager.getCameraIdList()[0]; // use the back camera
                    cameraManager.setTorchMode(cameraId, true);    // turn flashlight ON
                    btn.setText("Turn OFF");
                    isOn = true;
                } else {
                    cameraManager.setTorchMode(cameraId, false);   // turn flashlight OFF
                    btn.setText("Turn ON");
                    isOn = false;
                }
            } catch (CameraAccessException e) {
                // ignored for simplicity
            }
        });
    }
}
